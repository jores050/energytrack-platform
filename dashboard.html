/* ==================================================== ENERGYTRACK SaaS SUBSCRIPTION CHECK SYSTEM ==================================================== À METTRE DANS : dashboard.html JUSTE APRÈS : supabaseClient.auth.getUser() ==================================================== */ async function checkSubscriptionAccess(){ /* RÉCUPÉRER UTILISATEUR */ const { data:userData } = await supabaseClient.auth.getUser(); if(!userData.user){ window.location.href='auth.html'; return; } const userId = userData.user.id; /* RÉCUPÉRER ABONNEMENT */ const { data:subscription, error } = await supabaseClient .from('subscriptions') .select('*') .eq('user_id',userId) .single(); if(error || !subscription){ showUpgradePopup(); return; } /* DATE ACTUELLE */ const now = new Date(); /* DATE EXPIRATION */ const expiresAt = new Date(subscription.expires_at); /* ABONNEMENT EXPIRÉ */ if( subscription.status !== 'active' || expiresAt < now ){ showLimitedDashboard(); return; } /* ABONNEMENT ACTIF */ showFullDashboard(); } /* ==================================================== DASHBOARD COMPLET ==================================================== */ function showFullDashboard(){ console.log('Business Access Granted'); /* SECTIONS PREMIUM */ document .querySelectorAll('.premium-feature') .forEach(el=>{ el.style.display='block'; }); /* MESSAGE */ const banner = document.getElementById('subscriptionBanner'); if(banner){ banner.style.display='none'; } } /* ==================================================== MODE LIMITÉ GRATUIT ==================================================== */ function showLimitedDashboard(){ console.log('Limited Free Mode'); /* MASQUER FEATURES PREMIUM */ document .querySelectorAll('.premium-feature') .forEach(el=>{ el.style.display='none'; }); /* AFFICHER BANNIÈRE */ const banner = document.getElementById('subscriptionBanner'); if(banner){ banner.style.display='block'; banner.innerHTML = `

Votre abonnement a expiré

Vous utilisez actuellement le mode gratuit limité. Les fonctionnalités suivantes sont désactivées : • IA énergétique • Rapports PDF • Multi-sites • Analytics avancés • Alertes intelligentes

`; } } /* ==================================================== AUCUN ABONNEMENT ==================================================== */ function showUpgradePopup(){ const banner = document.getElementById('subscriptionBanner'); if(banner){ banner.style.display='block'; banner.innerHTML = `

Aucun abonnement détecté

Activez un abonnement Business Cloud pour débloquer toutes les fonctionnalités EnergyTrack.

`; } } /* ==================================================== LANCER VÉRIFICATION ==================================================== */ checkSubscriptionAccess();
