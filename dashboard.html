/* --- LOGIQUE DE VÉRIFICATION D'ABONNEMENT --- */ async function getSubscriptionStatus(userId) { try { const { data, error } = await supabaseClient .from('subscriptions') .select('status') .eq('user_id', userId) .eq('status', 'active') // On cherche spécifiquement un abonnement actif .maybeSingle(); // Retourne null proprement s'il n'y a rien if (error) throw error; // Si data existe et que le statut est 'active', on retourne true return !!(data && data.status === 'active'); } catch (err) { console.error("Erreur vérification abonnement:", err); return false; // Par défaut, on considère comme non-premium en cas d'erreur } } /* --- CHARGEMENT DU DASHBOARD --- */ async function loadDashboard() { const user = await getUser(); if (!user) return; // Appel de la nouvelle logique const isPremium = await getSubscriptionStatus(user.id); const container = document.getElementById('sitesContainer'); const headerPlaceholder = document.getElementById('premium-header-placeholder'); // 1. Gérer l'affichage de la bannière et le verrouillage du menu if (!isPremium) { headerPlaceholder.innerHTML = `
⚠️ Mode Gratuit Limité
Votre abonnement n'est plus actif. Accès restreint au site principal.
S'abonner maintenant
`; // Verrouiller les liens marqués 'data-premium' dans le menu document.querySelectorAll('.nav-item[data-premium="true"]').forEach(el => { el.classList.add('locked'); if (!el.querySelector('.premium-badge')) { el.innerHTML += ' PRO'; } el.href = "#"; el.onclick = (e) => { e.preventDefault(); alert("🔒 Cette fonctionnalité nécessite un abonnement actif."); }; }); } // 2. Charger les sites (Bridage SQL si non-premium) let query = supabaseClient.from('sites').select('*').eq('user_id', user.id); if (!isPremium) { query = query.limit(1); // On ne récupère que 1 site pour le mode gratuit } const { data: sites, error: sitesError } = await query; if (sitesError) return; container.innerHTML = ''; // Boucle de rendu des sites for (const site of sites) { // Récupération télémétrie (12 derniers points) const { data: telemetry } = await supabaseClient .from('telemetry') .select('*') .eq('site_id', site.id) .order('created_at', { ascending: false }) .limit(12); if (!telemetry || telemetry.length === 0) { container.innerHTML += `

${site.name}

Aucune donnée.

`; continue; } const latest = telemetry[0]; const totalKwh = telemetry.reduce((acc, curr) => acc + Number(curr.kwh_total || 0), 0); // Injection HTML de la carte container.innerHTML += `

🏢 ${site.name}

${site.city}
${latest.voltage > 50 ? 'SBEE OK' : 'COUPURE'}
Puissance
${Math.round(latest.power_w)} W
Conso. (session)
${totalKwh.toFixed(2)} kWh
`; // Initialisation différée du graphique setTimeout(() => renderChart(site.id, telemetry), 100); } // 3. Ajouter un bouton d'appel à l'action si limité if (!isPremium) { container.innerHTML += `

Vous avez d'autres sites ?

Débloquer le Multi-sites →
`; } } // Fonction utilitaire pour Chart.js function renderChart(siteId, data) { const ctx = document.getElementById(`chart-${siteId}`); if (!ctx) return; new Chart(ctx, { type: 'line', data: { labels: data.map(d => new Date(d.created_at).toLocaleTimeString()).reverse(), datasets: [{ data: data.map(d => d.power_w).reverse(), borderColor: '#00c896', backgroundColor: 'rgba(0, 200, 150, 0.05)', fill: true, tension: 0.4 }] }, options: { responsive: true, plugins: { legend: { display: false } }, scales: { x: { display: false }, y: { ticks: { color: '#64748b' } } } } }); } // Initialisation loadDashboard();
